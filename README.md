kNN <- function(xl, z, k)
{
  ## –°–æ—Ä—Ç–∏—Ä—É–µ–º –≤—ã–±–æ—Ä–∫—É —Å–æ–≥–ª–∞—Å–Ω–æ –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º–æ–≥–æ –æ–±—ä–µ–∫—Ç–∞

  n <- dim(orderedXl)[2] - 1
  
  ## –ü–æ–ª—É—á–∞–µ–º –∫–ª–∞—Å—Å—ã –ø–µ—Ä–≤—ã—Ö k —Å–æ—Å–µ–¥–µ–π
  classes <- orderedXl[1:k, n + 1]
  
  ## –°–æ—Å—Ç–∞–≤–ª—è–µ–º —Ç–∞–±–ª–∏—Ü—É –≤—Å—Ç—Ä–µ—á–∞–µ–º–æ—Å—Ç–∏ –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Å–∞
  counts <- table(classes)
  ## –ù–∞—Ö–æ–¥–∏–º –∫–ª–∞—Å—Å, –∫–æ—Ç–æ—Ä—ã–π –¥–æ–º–∏–Ω–∏—Ä—É–µ—Ç —Å—Ä–µ–¥–∏ –ø–µ—Ä–≤—ã—Ö k —Å–æ—Å–µ–¥–µ–π
  class <- names(which.max(counts))
  
  return (class)
}
```

naive = function(x, Py, mu, sigm, m, n) {
  gen<- matrix(c('setosa','versicolor', 'virginica', 0, 0, 0), nrow = 3, ncol = 2)
  scores = rep(0, m)
  for (i in 1:m) {
    scores[i] = Py[i]
    for (j in 1:n){
      N=1/sqrt(2*pi)/sigm[i,j]*exp(-1/2*(x[j]-mu[i,j])^2/sigm[i,j]^2)#–≤—ã—á–∏—Å–ª–µ–Ω–∏–µ –ø–ª–æ—Ç–Ω–æ—Å—Ç–µ–π
      scores[i] = scores[i] * N #–∏—â–µ–º –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Å–∞
    }
    gen[i,2]=scores[i]
  }
  class <- gen[,1][which.max(gen[,2])]
}
```
